📊 Propojení s Google Sheets
Záznamy se budou automaticky odesílat do vašeho spreadsheetu. Nastavení zabere ~5 minut.
1
Otevřete Google Sheets a vytvořte nový soubor s názvem "FC Docházka"
2
V menu klikněte na Rozšíření → Apps Script a vložte skript:
Tools → Paste Script
Skript si zkopírujte z pole níže ↓
3
Uložte a klikněte na Nasadit → Nové nasazení. Typ: Webová aplikace. Přístup: Kdokoliv. Zkopírujte URL a vložte sem ↓
Apps Script kód 
function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const data = JSON.parse(e.postData.contents);
  
  const sheetName = data.type === 'match' ? 'Zápasy' : 'Tréninky';
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    if (data.type === 'match') {
      sheet.appendRow(['Datum','Čas','Soupeř','Skóre','Přítomní','Omluvení','Nepřítomní','Jiné','Poznámka','Statistiky']);
    } else {
      sheet.appendRow(['Datum','Čas','Přítomní','Omluvení','Nepřítomní','Jiné','% Účast','Poznámka']);
    }
    sheet.getRange(1,1,1,sheet.getLastColumn()).setFontWeight('bold');
  }
  
  const present = data.players.filter(p=>p.status==='present').map(p=>p.name).join(', ');
  const excused = data.players.filter(p=>p.status==='excused').map(p=>p.name).join(', ');
  const absent = data.players.filter(p=>p.status==='absent').map(p=>p.name).join(', ');
  const other = data.players.filter(p=>p.status==='other').map(p=>p.name).join(', ');
  const pct = Math.round(data.players.filter(p=>p.status==='present').length / data.players.length * 100) + '%';
  
  if (data.type === 'match') {
    const stats = data.players.filter(p=>p.goals||p.assists).map(p=>`${p.name}(${p.goals}g/${p.assists}a)`).join(', ');
    sheet.appendRow([data.date, data.time, data.opponent||'', data.score||'', present, excused, absent, other, data.note||'', stats]);
  } else {
    sheet.appendRow([data.date, data.time, present, excused, absent, other, pct, data.note||'']);
  }
  
  return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);
}
